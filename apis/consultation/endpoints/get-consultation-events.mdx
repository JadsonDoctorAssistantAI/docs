---
title: 'Obter Eventos da Consulta'
api: 'GET https://apim.doctorassistant.ai/api/sandbox/consultations/{consultationId}/events'
description: 'Endpoint para obter o histórico de eventos de uma consulta específica'
---

## Autenticação

<ParamField header="x-daai-api-key" type="string" required>
  Chave de API para autenticação
</ParamField>

## Parâmetros da URL

<ParamField path="consultationId" type="string" required>
  ID da consulta
</ParamField>

## Resposta

### Resposta de Sucesso

<ResponseField name="status" type="integer">
  200 - Sucesso
</ResponseField>

<ResponseField name="data" type="array">
  <Expandable title="Toggle objeto">
    <ResponseField name="id" type="string">
      ID do evento
    </ResponseField>
    <ResponseField name="type" type="string">
      Tipo do evento
    </ResponseField>
    <ResponseField name="status" type="string">
      Status do evento
    </ResponseField>
    <ResponseField name="createdAt" type="string">
      Data de criação do evento
    </ResponseField>
    <ResponseField name="metadata" type="object">
      Metadados adicionais do evento
    </ResponseField>
  </Expandable>
</ResponseField>
