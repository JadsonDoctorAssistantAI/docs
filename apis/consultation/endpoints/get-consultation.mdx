---
title: 'Obter Consulta'
api: 'GET https://apim.doctorassistant.ai/api/sandbox/consultations/{consultationId}'
description: 'Endpoint para obter os detalhes de uma consulta específica'
---

## Autenticação

<ParamField header="x-daai-api-key" type="string" required>
  Chave de API para autenticação
</ParamField>

## Parâmetros da URL

<ParamField path="consultationId" type="string" required>
  ID da consulta
</ParamField>

## Resposta

### Resposta de Sucesso

<ResponseField name="status" type="integer">
  200 - Sucesso
</ResponseField>

<ResponseField name="data" type="object">
  <Expandable title="Toggle objeto">
    <ResponseField name="id" type="string">
      ID da consulta
    </ResponseField>
    <ResponseField name="professionalId" type="string">
      ID do profissional
    </ResponseField>
    <ResponseField name="status" type="string">
      Status atual da consulta
    </ResponseField>
    <ResponseField name="createdAt" type="string">
      Data de criação da consulta
    </ResponseField>
    <ResponseField name="updatedAt" type="string">
      Data da última atualização
    </ResponseField>
    <ResponseField name="metadata" type="object">
      Metadados adicionais da consulta
    </ResponseField>
    <ResponseField name="reason" type="string">
      Motivo da consulta
    </ResponseField>
    <ResponseField name="notes" type="string">
      Anotações da consulta
    </ResponseField>
  </Expandable>
</ResponseField>

### Exemplo de Resposta

```json
{
  "status": 200,
  "data": {

  }

```
